# Lambda deployment requirements for document ingestion pipeline
#
# This file is the SOURCE for dependency resolution.
# The Dockerfile uses requirements.lock (not this file) for installation.
#
# To regenerate the lockfile with pinned versions and hashes:
#   uv pip compile requirements.txt -o requirements.lock --generate-hashes --python-version 3.12 \
#     --no-emit-package torch --no-emit-package torchvision
#
# torch and torchvision are excluded because the Dockerfile installs torch+cpu
# from the PyTorch index separately (step 4). Including them here would cause
# --require-hashes to fail due to the +cpu variant mismatch.
#
# To build Lambda package manually:
#   uv pip install -r requirements.lock --require-hashes --target ./package
#   zip -r lambda-package.zip package/

# NumPy and SciPy - AL2023 has glibc 2.34 so modern wheels work fine
numpy
scipy

# AWS SDK (already in Lambda runtime, but pinning for consistency)
boto3>=1.40.1

# Document processing libraries
# transformers 4.45+ has RTDetrImageProcessor needed by docling
# We install torch CPU explicitly in Dockerfile before this
transformers>=4.45.0  # Required for RTDetrImageProcessor used by docling
docling>=1.0.0  # Core Docling library for PDF, DOCX, PPTX, TXT, MD, RTF processing
docling-core[chunking-openai]  # Chunking with OpenAI tokenizer
tiktoken>=0.5.0  # Text chunking and tokenization
striprtf>=0.0.29  # RTF to text conversion
